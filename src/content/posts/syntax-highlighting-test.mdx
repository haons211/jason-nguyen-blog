## TypeScript

```typescript
interface User {
  id: number;
  name: string;
  email: string;
  isActive: boolean;
}

type UserRole = 'admin' | 'user' | 'moderator';

class UserService {
  private users: User[] = [];

  constructor(private apiUrl: string) {}

  async createUser(userData: Omit<User, 'id'>): Promise<User> {
    const newUser: User = {
      ...userData,
      id: Date.now()
    };
    
    this.users.push(newUser);
    return newUser;
  }

  findUsersByRole<T extends UserRole>(role: T): User[] {
    return this.users.filter(user => user.role === role);
  }
}
```

## Bash

```bash
#!/bin/bash

# Variables
PROJECT_NAME="my-app"
BUILD_DIR="dist"
LOG_FILE="deploy.log"

# Functions
log_message() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
}

deploy_app() {
    local environment=$1
    
    log_message "Starting deployment to $environment"
    
    # Build the project
    if npm run build; then
        log_message "Build successful"
    else
        log_message "Build failed"
        exit 1
    fi
    
    # Deploy based on environment
    case $environment in
        "production")
            rsync -avz "$BUILD_DIR/" user@prod-server:/var/www/html/
            ;;
        "staging")
            rsync -avz "$BUILD_DIR/" user@staging-server:/var/www/html/
            ;;
        *)
            log_message "Unknown environment: $environment"
            exit 1
            ;;
    esac
    
    log_message "Deployment to $environment completed"
}

# Main execution
if [ $# -eq 0 ]; then
    echo "Usage: $0 <environment>"
    echo "Available environments: production, staging"
    exit 1
fi

deploy_app "$1"
```
